IOS运用CGContextRef画支付框

首先了解一下 CGContextRef ：
An opaque type that represents a Quartz 2D drawing environment.
Graphics Context是图形上下文,可以将其理解为一块画布,我们可以在上面进行绘画操作,绘制完成后,将画布放到我们的view中显示即可,view看作是一个画框.

自定义PayPassWordView类

#import <UIKit/UIKit.h>

@interface PayPassWordView : UIView
//设置输入的支付密码
@property (nonatomic, strong)NSString *str;

//设置圆角

@property (nonatomic, assign)CGFloat cornerRadius;
@end

实现PayPassWordView类

#import "PayPassWordView.h"

@implementation PayPassWordView


- (void)drawRect:(CGRect)rect {

[super drawRect:rect];

//An opaque type that represents a Quartz 2D drawing environment.
//一个不透明类型的Quartz 2D绘画环境,相当于一个画布,你可以在上面任意绘画  
CGContextRef context = UIGraphicsGetCurrentContext();

//计算密码框的宽度
float fw = ScreenWidth() - ViewWidth(102);
//计算密码框的宽度
float fh = (ScreenWidth() - ViewWidth(102))/6;
//圆角的大小
CGFloat f = ViewWidth(_cornerRadius);
// 开始坐标右边开始
CGContextMoveToPoint(context, fw, fh - f);  
// 绘制右下角圆角
CGContextAddArcToPoint(context, fw, fh, fw - f, fh, f / 2); 
// 绘制左下角圆角
CGContextAddArcToPoint(context, ViewWidth(1), fh, ViewWidth(1) , fh - f , f / 2);
// 绘制左上角圆角
CGContextAddArcToPoint(context, ViewWidth(1) , ViewWidth(1),  f, ViewWidth(1), f / 2); 
// 绘制右上角圆角
CGContextAddArcToPoint(context, fw, ViewWidth(1), fw, f, f / 2); 
// 将圆角矩形封起来
CGContextClosePath(context);

//绘制第一条分割线
CGContextMoveToPoint(context, fh, ViewWidth(1));
CGContextAddLineToPoint(context, fh,fh);
//绘制第二条分割线
CGContextMoveToPoint(context,2 * fh, ViewWidth(1));
CGContextAddLineToPoint(context, 2 * fh,fh);
//绘制第仨条分割线
CGContextMoveToPoint(context, 3 * fh, ViewWidth(1));
CGContextAddLineToPoint(context, 3 * fh,fh);
//绘制第四条分割线
CGContextMoveToPoint(context, 4 * fh, ViewWidth(1));
CGContextAddLineToPoint(context, 4 * fh,fh);
//绘制第五条分割线
CGContextMoveToPoint(context, 5 * fh, ViewWidth(1));
CGContextAddLineToPoint(context, 5 * fh,fh);

// 设置画笔颜色
CGContextSetRGBStrokeColor(context, 0.5, 0.5, 0.5, 0.5);
//根据上述坐标绘制路径
CGContextStrokePath(context);

for (NSInteger i = 0 ; i<_str.length; i++) {
//"根据支付密码位数，绘制黑色小圆点
CGContextFillEllipseInRect(context, CGRectMake(fh / 2 - 5 + i*fh, fh / 2 - 5, 10, 10));
}
}

//实现str的set方法

- (void)setStr:(NSString *)str
{
//设置输入的支付密码
_str = str;
//重绘界面
[self setNeedsDisplay];
}

@end
