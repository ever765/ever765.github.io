UIScrollView上下联动MenuView的实现

定义头部 MenuView

#import <UIKit/UIKit.h>

@class MenuView;

@protocol MenuViewDelegate <NSObject>

- (void)menuView:(MenuView *)menuView didSelectIndex:(NSInteger)index;

@end

@interface MenuView : UIScrollView

//每个按钮的title
@property (nonatomic, strong)NSArray *titleNames;

//每个按钮的颜色
@property (nonatomic, strong)UIColor *selectColor;

//每个按钮的颜色
@property (nonatomic, strong)UIColor *normalColor;

//每个按钮之间的间距   默认10px
@property (nonatomic, assign)CGFloat buttonSpace;
//设置背景颜色
@property (nonatomic, strong)UIColor *backColor;
//按钮titleLabel 字体大小
@property (nonatomic, assign)CGFloat fontSize;
//按钮的宽度
@property (nonatomic, assign)CGFloat buttonWidth;
//设置代理
@property (nonatomic, weak)id <MenuViewDelegate> menuDelegate;
//选中的按钮
- (void)selectButtonWithIndex:(NSInteger)index;

@end

实现头部MenuView

#import "MenuView.h"
#import "CustomButton.h"

@interface MenuView (){
NSMutableArray *_buttons;
CustomButton *_lastButton;
}

@end

@implementation MenuView

- (instancetype)initWithFrame:(CGRect)frame{
self = [super initWithFrame:frame];
if (self) {
_buttonSpace = 10;
_selectColor = [UIColor redColor];
_normalColor = [UIColor blackColor];
_backColor = UIColorFromRGB(0xeeeeee);
_buttons = [[NSMutableArray alloc] init];
_fontSize = 13;

}
return self;
}

- (void)setTitleNames:(NSArray *)titleNames{

_titleNames = titleNames;
self.backgroundColor = _backColor;
for (NSInteger index = 0; index < _titleNames.count; index++) {
CustomButton *button = [CustomButton buttonWithType:UIButtonTypeCustom];
[button setTitle:titleNames[index] forState:UIControlStateNormal];
button.frame = [self buttonFrameWithIndex:index];
[button setTitleColor:_normalColor forState:UIControlStateNormal];
[button setTitleColor:_selectColor forState:UIControlStateSelected];
button.selectColor = _selectColor;
button.titleLabel.font = [UIFont systemFontOfSize:_fontSize];
button.tag = index;
[button addTarget:self action:@selector(buttonAction:) forControlEvents:UIControlEventTouchDown];
[self addSubview:button];
if (index == 0) {
_lastButton = button;
_lastButton.selected = YES;
}
[_buttons addObject:button];
if (index == _titleNames.count - 1) {
self.contentSize = CGSizeMake(CGRectGetMaxX(button.frame), VIEWHEIGHT);
self.showsHorizontalScrollIndicator = NO;
self.showsVerticalScrollIndicator = NO;
}
}
}

- (void)selectButtonWithIndex:(NSInteger)index
{
_lastButton.selected = NO;
_lastButton = _buttons[index];
_lastButton.selected = YES;
if (_lastButton.frame.origin.x > SCREENWIDTH - _lastButton.frame.size.width + _buttonSpace) {
self.contentOffset = CGPointMake(_lastButton.frame.origin.x + _lastButton.frame.size.width + _buttonSpace - SCREENWIDTH , 0);
}else{
self.contentOffset = CGPointZero;
}
}

- (void)buttonAction:(CustomButton *)button
{
if (![_lastButton isEqual:button]) {
_lastButton.selected = NO;
_lastButton = button;
_lastButton.selected = YES;
if ([_menuDelegate respondsToSelector:@selector(menuView:didSelectIndex:)]) {
[_menuDelegate menuView:self didSelectIndex:button.tag];
}
}
}

- (CGRect)buttonFrameWithIndex:(NSInteger)index{
CGRect rect;

if (_buttonWidth) {
rect =  CGRectMake(index  * (_buttonWidth +_buttonSpace) + _buttonSpace, VIEWHEIGHT / 4, _buttonWidth, VIEWHEIGHT / 2);
}else{
rect = CGRectMake(index * ((SCREENWIDTH - _titleNames.count * _buttonSpace)/_titleNames.count +_buttonSpace) + _buttonSpace, VIEWHEIGHT / 4, (SCREENWIDTH - _titleNames.count * _buttonSpace)/_titleNames.count, VIEWHEIGHT / 2);
}
return rect;
}
